# Kinetic equations
alpha= (10.0 / (1.0 + exp( - (0.1 * (6.0 + V)))))

a= (k1 * k3_ / (k1_ * k3))^0.5
O= A_na^3

inhibitedA'=A_na*k1+inhibitedD*alpha*a-inhibitedA*(k1_+beta*a)
inhibitedD'=inhibitedA*beta*a+D*k3-inhibitedD*(alpha*a+k3_)

rIR' = (rIRinf - rIR)/tauIRr
h2'= kio*(1-h2)-koi*ACO*n^4*h2
